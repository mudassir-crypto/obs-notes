##### RequestParam
```java

@GetMapping("/products")
public ResponseEntity<Map<String, Object>> getProducts(
	@RequestParam("name") String productName,
	@RequestParam("id") String id,
	@RequestParam(value = "rating", defaultValue = "3", required = false) int rating
){
	Map<String, Object> response = new HashMap<>();
	response.put("name", productName);
	response.put("id", id);
	response.put("rating", rating);
	return ResponseEntity.ok(response);
```

TodoService:
```java
package com.linx.todo.services;

import java.util.ArrayList;
import java.util.List;
import java.util.stream.Collectors;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Component;

import com.linx.todo.models.Todo;

@Component
public class TodoService {
	
	Logger logger = LoggerFactory.getLogger(TodoService.class);
	
	private List<Todo> todos = new ArrayList<>();
	
	public Todo createTodo(Todo todo) {
		todos.add(todo);
		return todo;
	}
	
	public List<Todo> getTodos(){
		return this.todos;
	}
	
	public Todo getTodo(int id) {
		Todo todo = todos.stream().filter(t -> t.getId() == id).findAny().get();
		logger.info("getSingleTodo: {}", todo);;
		return todo;
	}
	
	public Todo updateTodo(int id, Todo todo) {
		List<Todo> updatedTodos = todos.stream().map(t -> {
			if(t.getId() == id) {
				t.setTitle(todo.getTitle());
				t.setContent(todo.getContent());
				t.setStatus(todo.getStatus());
				return t;
			} else {
				return t;
			}
		}).collect(Collectors.toList());
		this.todos = updatedTodos;
		todo.setId(id);
		return todo;
	}
	
	public void deleteTodo(int id) {
		List<Todo> newTodos = todos.stream().filter(t -> t.getId() != id).collect(Collectors.toList());
		this.todos = newTodos;
	}
}

```
